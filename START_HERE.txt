Полнофункциональная прошивка контроллера умного дома на Python с поддержкой:

✅ Modbus RTU        - Взаимодействие с RTU устройствами через serial порт
✅ Modbus TCP        - Доступ к RTU через TCP сеть
✅ Веб интерфейс    - Современный HTML/CSS/JS интерфейс управления
✅ REST API          - 14+ API endpoints для программного управления
✅ Конфигурация     - JSON-based конфигурация с автосохранением
✅ Сетевые параметры - Управление hostname, IP адресом, DHCP
✅ Логирование      - Подробное логирование всех операций
✅ Тесты            - Unit тесты для проверки функциональности
✅ Документация     - Полная документация на русском и английском

🚀 БЫСТРЫЙ СТАРТ (Выберите способ)
═══════════════════════════════════════════════════════════════════════════════

СПОСОБ 1️⃣ - Разработка на Windows
────────────────────────────────────
1. python -m venv venv
2. venv\Scripts\activate
3. pip install -r requirements.txt
4. python main.py
5. Открыть http://localhost:8000

СПОСОБ 2️⃣ - Ubuntu (Рекомендуется для продакшена)
────────────────────────────────────────────────────
1. sudo chmod +x install.sh
2. sudo ./install.sh
3. sudo systemctl start smarthome
4. Открыть http://<IP>:8000

СПОСОБ 3️⃣ - Docker (Контейнер)
────────────────────────────────
1. docker-compose up -d
2. Открыть http://localhost:8000

Через веб-интерфейс (http://localhost:8000):
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  📍 Вкладка "Сеть":
     • Установить/получить имя хоста
     • Просмотреть сетевые интерфейсы и IP адреса
  
  📍 Вкладка "Modbus RTU":
     • Подключиться к RTU устройствам
     • Выбрать порт и скорость передачи
     • Читать/писать регистры и катушки
     • Просмотреть результаты операций
  
  📍 Вкладка "Modbus TCP":
     • Запустить/остановить TCP сервер
     • Настроить IP и порт
     • Просмотреть статус сервера
  
  📍 Вкладка "Конфигурация":
     • Просмотреть и отредактировать JSON конфигурацию
     • Сохранить изменения

Через REST API (примеры с curl):
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  # Статус системы
  curl http://localhost:8000/api/system/status
  
  # Чтение регистров
  curl -X POST http://localhost:8000/api/modbus/rtu/read \
    -H "Content-Type: application/json" \
    -d '{"slave_id": 1, "type": "holding_registers", "start_addr": 0, "quantity": 10}'
  
  # Запись регистра
  curl -X POST http://localhost:8000/api/modbus/rtu/write \
    -H "Content-Type: application/json" \
    -d '{"slave_id": 1, "type": "register", "addr": 0, "value": 100}'
  
  # Запуск TCP сервера
  curl -X POST http://localhost:8000/api/modbus/tcp/start \
    -H "Content-Type: application/json" \
    -d '{"host": "0.0.0.0", "port": 5020}'

Через Python клиент:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  from examples.python_client import SmartHomeClient
  
  client = SmartHomeClient("http://localhost:8000/api")
  
  # Подключиться к RTU
  client.rtu_connect("/dev/ttyUSB0", 9600)
  
  # Прочитать регистры
  data = client.read_holding_registers(slave_id=1, start_addr=0, quantity=10)
  print(data)
  
  # Запустить TCP сервер
  client.tcp_start()
